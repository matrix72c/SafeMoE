    ```python
    # safemoe/data/datamodule.py
    import itertools
    import random
    from dataclasses import dataclass, field
    from pathlib import Path
    from typing import Optional

    from litgpt.data import DataModule
    from litgpt.tokenizer import Tokenizer


    @dataclass
    class MultiDataLoader(DataModule):
        """Multi-split DataLoader for SGTM training.

        Serves batches from D_std, D_harmful, and D_unlabeled splits using
        dynamic weighted sampling. Each call to next() returns (batch, split_label).

        Cache directory layout (produced by safemoe/data/prepare.py):
            {cache_dir}/{tokenizer.model_name}/x{x_harmful_unlabeled}/{split}/{train|val}
        """
        cache_dir: Path = Path("data/.cache")
        x_harmful_unlabeled: float = 0.0
        upsample_std: float = 1.0
        upsample_harmful: float = 1.0
        upsample_unlabeled: float = 1.0
        seed: int = 42
        num_workers: int = 4

        # Set via connect() — not init args
        tokenizer: Optional[object] = field(default=None, init=False, repr=False)
        batch_size: int = field(default=1, init=False, repr=False)
        max_seq_length: int = field(default=-1, init=False, repr=False)

        def __post_init__(self):
            super().__init__()

        def connect(
            self,
            tokenizer=None,
            batch_size: int = 1,
            max_seq_length: int = -1,
            **kwargs,
        ) -> None:
            self.tokenizer = tokenizer
            self.batch_size = batch_size
            self.max_seq_length = max_seq_length + 1  # +1 for next-token prediction target

        def setup(self, stage: str = "") -> None:
            from litdata.streaming import StreamingDataLoader, StreamingDataset, TokensLoader

            random.seed(self.seed)

            tokenizer_name = self.tokenizer.model_name if self.tokenizer else "default"
            base = Path(self.cache_dir) / tokenizer_name / f"x{self.x_harmful_unlabeled}"

            def make_train_loader(split_name: str) -> StreamingDataLoader:
                ds = StreamingDataset(
                    input_dir=str(base / split_name / "train"),
                    item_loader=TokensLoader(block_size=self.max_seq_length),
                    shuffle=True,
                )
                return StreamingDataLoader(
                    ds,
                    batch_size=self.batch_size,
                    num_workers=self.num_workers,
                    pin_memory=True,
                    drop_last=True,
                )

            self._loaders = {
                "D_std":       make_train_loader("D_std"),
                "D_harmful":   make_train_loader("D_harmful"),
                "D_unlabeled": make_train_loader("D_unlabeled"),
            }

            # Infinite cycling iterators per split
            self._iters = {k: itertools.cycle(v) for k, v in self._loaders.items()}

            # Sampling weights (upsample factors only — split sizes already reflected in datasets)
            self._split_names = ["D_std", "D_harmful", "D_unlabeled"]
            self._weights = [
                self.upsample_std,
                self.upsample_harmful,
                self.upsample_unlabeled,
            ]

        def next(self):
            """Draw one batch from a weighted-randomly chosen split.

            Returns:
                (batch, split_label): batch is a torch.Tensor of shape
                    (batch_size, max_seq_length+1); split_label is one of
                    "D_std", "D_harmful", "D_unlabeled".
            """
            split = random.choices(self._split_names, weights=self._weights, k=1)[0]
            batch = next(self._iters[split])
            return batch, split

        def val_dataloaders(self) -> dict:
            """Return per-split validation DataLoaders.

            D_unlabeled has no validation set (per paper reference implementation).

            Returns:
                dict with keys "D_std" and "D_harmful", each mapping to a DataLoader.
            """
            from litdata.streaming import StreamingDataLoader, StreamingDataset, TokensLoader

            tokenizer_name = self.tokenizer.model_name if self.tokenizer else "default"
            base = Path(self.cache_dir) / tokenizer_name / f"x{self.x_harmful_unlabeled}"

            def make_val_loader(split_name: str) -> StreamingDataLoader:
                ds = StreamingDataset(
                    input_dir=str(base / split_name / "val"),
                    item_loader=TokensLoader(block_size=self.max_seq_length),
                    shuffle=False,  # Validation must be deterministic
                )
                return StreamingDataLoader(
                    ds,
                    batch_size=self.batch_size,
                    num_workers=self.num_workers,
                    pin_memory=True,
                    drop_last=False,
                )

            return {
                "D_std":     make_val_loader("D_std"),
                "D_harmful": make_val_loader("D_harmful"),
            }

        def train_dataloader(self):
            """Standard DataModule interface.

            Returns the D_std loader. The SGTM training loop uses next() instead;
            this method exists for LightningDataModule compatibility.
            """
            return self._loaders["D_std"]

        def val_dataloader(self):
            """Standard DataModule interface.

            Returns list of val loaders. The SGTM eval loop uses val_dataloaders()
            instead; this method exists for LightningDataModule compatibility.
            """
            return list(self.val_dataloaders().values())
    ```

    Note on `test_seed_reproducibility`: The test creates two separate `MultiDataLoader` instances with the same seed and checks that 20 consecutive next() calls produce the same split sequence. This works because `setup()` calls `random.seed(self.seed)` which resets the global random state. IMPORTANT: This means creating a second loader WILL reset the global random state for the first loader's ongoing sequence. For the test, both instances are fresh and each runs 20 calls from a fresh seed — this is the intended behavior.

    Note on `num_workers=0` in tests: The test fixture passes `num_workers=0` for compatibility with the test environment. `StreamingDataLoader` with `num_workers=0` runs synchronously. Ensure the implementation does not error when `num_workers=0`.

    Note on `test_sampling_weights`: Uses `n=300` calls with `upsample_harmful=3.0` vs `1.0` for std/unlabeled. Expected D_harmful count = `300 * 3/(1+3+1) = 180`. Tolerance is 20% = 36 calls. With 300 samples this should pass reliably (law of large numbers).
