       n_en = len(en)
       n_es = len(es)
       d_std = en[:int(0.25 * n_en)]
       d_unlabeled_en = en[int(0.25 * n_en):]
       n_harmful = int((1.0 - x / 100.0) * n_es)

    2. `tokenize_stories(stories: list, tokenizer)` — generator for litdata.optimize().
       Do NOT read env vars (no `DATA_OPTIMIZER_GLOBAL_RANK`). Simple loop:
       ```python
       def tokenize_stories(stories: list, tokenizer):
           for text in stories:
               text = text.strip()
               if not text:
                   continue
               tokens = tokenizer.encode(text, bos=True, eos=False)
               yield tokens
       ```

    3. `_maybe_optimize(stories, output_dir, tokenizer, num_workers, chunk_bytes)` — idempotent wrapper:
       ```python
       def _maybe_optimize(stories, output_dir, tokenizer, num_workers=4, chunk_bytes="200MB"):
           from litdata import TokensLoader, optimize
           from functools import partial
           output_dir = Path(output_dir)
           if output_dir.is_dir():
               print(f"Skipping {output_dir} (already exists)")
               return
           output_dir.mkdir(parents=True, exist_ok=True)
           # Split stories across workers for parallelism
           worker_inputs = [stories[i::num_workers] for i in range(num_workers)] if num_workers > 1 else [stories]
           optimize(
               fn=partial(tokenize_stories, tokenizer=tokenizer),
               inputs=worker_inputs,
               output_dir=str(output_dir),
               num_workers=num_workers,
               chunk_bytes=chunk_bytes,
               item_loader=TokensLoader(),
           )
       ```

    4. `prepare(en_train, es_train, en_val, es_val, cache_dir, tokenizer, x, num_workers=4, chunk_bytes="200MB")`
       ```python
       def prepare(en_train, es_train, en_val, es_val, cache_dir, tokenizer,
                   x=0.0, num_workers=4, chunk_bytes="200MB"):
           splits = compute_splits(en_train, es_train, x)
           base = Path(cache_dir) / tokenizer.model_name / f"x{x}"
           _maybe_optimize(splits["D_std"],       base / "D_std/train",     tokenizer, num_workers, chunk_bytes)
           _maybe_optimize(en_val,                base / "D_std/val",       tokenizer, 1, chunk_bytes)
           _maybe_optimize(splits["D_harmful"],   base / "D_harmful/train", tokenizer, num_workers, chunk_bytes)
           _maybe_optimize(es_val,                base / "D_harmful/val",   tokenizer, 1, chunk_bytes)
           _maybe_optimize(splits["D_unlabeled"], base / "D_unlabeled/train", tokenizer, num_workers, chunk_bytes)
           # No D_unlabeled/val
       ```
