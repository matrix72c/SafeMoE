```python
from functools import partial

optimize(
    fn=partial(tokenize_fn, tokenizer=tokenizer),
    inputs=[sublist_per_worker],   # each worker gets a slice
    output_dir=str(output_dir),
    num_workers=4,
    chunk_bytes="200MB",
    item_loader=TokensLoader(),
)

# tokenize function — avoid DATA_OPTIMIZER_* env var reads (KeyError when num_workers=1)
def tokenize_stories(stories: list[str], tokenizer: Tokenizer):
    for text in stories:
        text = text.strip()
        tokens = tokenizer.encode(text, bos=True, eos=False)
        yield tokens
```
