# ---------------------------------------------------------------------------
# EVAL-03: evaluate_with_ablation() — in-training ablation evaluation

from safemoe.pretrain import evaluate_with_ablation  # RED: not yet defined


class _MockValLoadersForPretrain:
    """Minimal val loaders dict for evaluate_with_ablation tests."""

    def val_dataloaders(self) -> dict:
        ds = _SynthDataset(5, 32, 1024)
        return {
            "D_std": DataLoader(ds, batch_size=1, drop_last=False),
            "D_harmful": DataLoader(ds, batch_size=1, drop_last=False),
        }


def test_evaluate_with_ablation_restores_weights():
    """evaluate_with_ablation() must restore theta_harmful weights after ablation eval."""
    model, registry, config = _build_model_and_registry()
    fabric = L.Fabric(devices=1, accelerator="cpu", strategy="auto")
    fabric.launch()
    model = fabric.setup(model)

    harmful_params = registry.parameters_by_type("theta_harmful")
    # Snapshot original values
    originals = [p.data.clone() for p in harmful_params]

    val_loaders = _MockValLoadersForPretrain().val_dataloaders()
    evaluate_with_ablation(fabric, model, registry, val_loaders, iter_num=10)

    # All theta_harmful params must be restored to original values
    for orig, p in zip(originals, harmful_params):
        assert torch.allclose(orig, p.data), (
            "evaluate_with_ablation left model in ablated state — weights not restored"
        )


def test_evaluate_with_ablation_logs_metrics():
    """evaluate_with_ablation() must log ablated_val_ppl_D_std and _D_harmful only."""
    model, registry, config = _build_model_and_registry()
    fabric = L.Fabric(devices=1, accelerator="cpu", strategy="auto")
    fabric.launch()
    model = fabric.setup(model)

    logged_metrics: dict = {}

    def _capture_log_dict(metrics, step=None):
        logged_metrics.update(metrics)

    fabric.log_dict = _capture_log_dict  # type: ignore[method-assign]

    val_loaders = _MockValLoadersForPretrain().val_dataloaders()
    evaluate_with_ablation(fabric, model, registry, val_loaders, iter_num=10)

    assert "ablated_val_ppl_D_std" in logged_metrics, (
        "evaluate_with_ablation must log ablated_val_ppl_D_std"
    )
    assert "ablated_val_ppl_D_harmful" in logged_metrics, (
        "evaluate_with_ablation must log ablated_val_ppl_D_harmful"
    )
    # User decision: D_unlabeled metrics are skipped
    assert "ablated_val_ppl_D_unlabeled" not in logged_metrics, (
        "evaluate_with_ablation must NOT log D_unlabeled metrics (user decision)"
    )
