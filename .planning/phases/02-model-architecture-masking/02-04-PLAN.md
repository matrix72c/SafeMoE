    class ActivationMasker:
        def __init__(self, model: nn.Module, registry: HarmfulParamRegistry) -> None:

            from safemoe.model import SafeMoELayer
            self._layers: list[SafeMoELayer] = [

python -c "
import torch
import litgpt
from safemoe.config import SafeMoEConfig
from safemoe.model import SafeMoELayer
from safemoe.masking import HarmfulParamRegistry, GradientMasker, ActivationMasker

cfg = SafeMoEConfig(padded_vocab_size=1024, vocab_size=1024, n_layer=2, n_embd=32, n_head=4, n_query_groups=4, head_size=8, n_expert=4, n_expert_per_token=2, moe_intermediate_size=64, mlp_class_name='LLaMAMoE', harmful_expert_indices=[0,1], num_harmful_experts=2, harmful_attn_heads=[])
model = litgpt.GPT(cfg)
registry = HarmfulParamRegistry(model, cfg)
grad_masker = GradientMasker(registry)
act_masker = ActivationMasker(model, registry)

# D_harmful step
x = torch.randint(0, 1024, (1, 4))
loss = model(x).sum()
loss.backward()
grad_masker.mask()
assert all(p.grad is None for p in registry.parameters_by_type('theta_std')), 'theta_std grads must be None'
assert all(p.grad is not None for p in registry.parameters_by_type('theta_harmful')), 'theta_harmful grads must be non-None'
print('OK: GradientMasker invariant')

# D_std step
model.zero_grad(set_to_none=True)
act_masker.enable()
out = model(x)
act_masker.disable()
for m in model.modules():
    if isinstance(m, SafeMoELayer):
        assert not m._activation_masking_enabled
print('OK: ActivationMasker enable/disable invariant')
"
