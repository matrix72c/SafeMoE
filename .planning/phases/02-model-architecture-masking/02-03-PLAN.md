    Implement HarmfulParamRegistry:
    ```python
    class HarmfulParamRegistry:
        def __init__(self, model: nn.Module, config: "SafeMoEConfig") -> None:
            harmful_expert_indices = set(config.harmful_expert_indices)
            harmful_attn_heads = list(getattr(config, 'harmful_attn_heads', []))
            theta_harmful: list[nn.Parameter] = []
            theta_std: list[nn.Parameter] = []
            # Stores (qkv_param, head_slice_indices) for Phase 3 attention head masking.
            # head_slice_indices is a list of slice objects covering the Q/K/V rows
            # for each head index in harmful_attn_heads (MHA layout assumed for Phase 2).
            qkv_harmful_metadata: list[tuple[nn.Parameter, list[slice]]] = []

            # Pattern: transformer.h.{layer}.mlp.experts.{idx}.{anything}
            expert_pattern = re.compile(
                r"transformer\.h\.\d+\.mlp\.experts\.(\d+)\."
            )
            # Pattern: transformer.h.{layer}.attn.qkv.weight
            qkv_pattern = re.compile(
                r"transformer\.h\.(\d+)\.attn\.qkv\.weight$"
            )

            for name, param in model.named_parameters():
                m_expert = expert_pattern.match(name)
                m_qkv = qkv_pattern.match(name)

                if m_expert and int(m_expert.group(1)) in harmful_expert_indices:
                    # Full expert parameter -> theta_harmful (masker-ready)
                    theta_harmful.append(param)
                elif m_qkv and harmful_attn_heads:
                    # Full qkv.weight -> theta_std for Phase 2 masker use.
                    # Also store (param, row_slices) metadata for Phase 3.
                    # Row slice math (MHA: n_query_groups == n_head):
                    #   Q rows for head i: [i*head_size : (i+1)*head_size]
                    #   K rows for head i: [n_head*head_size + i*head_size : n_head*head_size + (i+1)*head_size]
                    #   V rows for head i: [(n_head + n_query_groups)*head_size + i*head_size : ...]
                    n_head = config.n_head
                    n_query_groups = config.n_query_groups
                    head_size = config.head_size
                    slices: list[slice] = []
                    for head_idx in harmful_attn_heads:
                        q_start = head_idx * head_size
                        slices.append(slice(q_start, q_start + head_size))
                        # K/V use n_query_groups stride (handles both MHA and GQA)
                        kv_head = head_idx % n_query_groups
                        k_start = n_head * head_size + kv_head * head_size
                        slices.append(slice(k_start, k_start + head_size))
                        v_start = (n_head + n_query_groups) * head_size + kv_head * head_size
                        slices.append(slice(v_start, v_start + head_size))
                    qkv_harmful_metadata.append((param, slices))
                    # qkv.weight goes into theta_std (full parameter, not sliced)
                    theta_std.append(param)
                else:
                    theta_std.append(param)

            # Exhaustive + non-overlapping validation using id()
            harmful_ids = {id(p) for p in theta_harmful}
            std_ids = {id(p) for p in theta_std}
            all_ids = {id(p) for _, p in model.named_parameters()}

            if harmful_ids & std_ids:
                raise ValueError(
                    "HarmfulParamRegistry: theta_harmful and theta_std overlap — "
                    f"{len(harmful_ids & std_ids)} parameter(s) in both sets."
                )
            if harmful_ids | std_ids != all_ids:
                missing = all_ids - (harmful_ids | std_ids)
                raise ValueError(
                    "HarmfulParamRegistry: not all parameters classified — "
                    f"{len(missing)} parameter(s) unclassified."
                )

            self._registry: dict[str, list[nn.Parameter]] = {
                "theta_harmful": theta_harmful,
                "theta_std": theta_std,
            }
            # Phase 3 uses this to apply row-level gradient masking on qkv.weight.
            # Each entry: (qkv_weight_param, list_of_row_slices_for_harmful_heads).
            self._qkv_harmful_metadata: list[tuple[nn.Parameter, list[slice]]] = (
                qkv_harmful_metadata
            )

        def parameters_by_type(self, split: str) -> list[nn.Parameter]:
            if split not in self._registry:
                raise KeyError(f"Unknown split '{split}'; expected 'theta_harmful' or 'theta_std'")
            return self._registry[split]
    ```

    Note on duality: qkv.weight is added to theta_std (so exhaustive coverage holds and the Phase 2
    maskers never touch it), but the (param, slices) tuple is ALSO stored in _qkv_harmful_metadata.
    This is intentional: the full parameter identity stays in theta_std for set-cover validation;
    the metadata is separate so Phase 3 can do row-level gradient zeroing on just the harmful rows.
