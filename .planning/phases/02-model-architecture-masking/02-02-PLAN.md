```python
def forward(self, x: torch.Tensor) -> torch.Tensor:
    B, T, C = x.size()
    x = x.view(-1, C)
    router = self.gate(x)
    probs, indices = torch.topk(router, self.config.n_expert_per_token)
    probs = probs.softmax(dim=1, dtype=torch.float).to(dtype=x.dtype)
    masks = indices.unsqueeze(-1) == torch.arange(self.config.n_expert, device=x.device)
    masks = masks.permute(2, 0, 1)
    y = torch.zeros_like(x)
    for mask, expert in zip(masks, self.experts):
        token_idx, expert_idx = torch.where(mask)
        y[token_idx] += probs[token_idx, expert_idx, None] * expert(x[token_idx])
    return y.view(B, T, C)
```

    Implement SafeMoELayer:
    ```python
    class SafeMoELayer(LLaMAMoE):
        def __init__(self, config: SafeMoEConfig) -> None:
            super().__init__(config)
            self._activation_masking_enabled: bool = False
            self._harmful_indices: list[int] = list(config.harmful_expert_indices)
            # Apply copy-weights init strategy if requested
            # init_strategy stored on config or passed separately
            # Per CONTEXT.md: "configurable (random init vs copy weights from existing std experts)"
            # Add init_strategy field support: check config.harmful_expert_init if it exists,
            # or accept it as a constructor arg. Simplest: add optional harmful_expert_init: str = "random"
            # to SafeMoEConfig (update config.py accordingly), default "random" = no action.
            # For "copy": find first std expert index (not in harmful_expert_indices), deepcopy its
            # state_dict into each harmful expert.
            if getattr(config, 'harmful_expert_init', 'random') == 'copy':
                std_indices = [i for i in range(len(self.experts))
                               if i not in self._harmful_indices]
                if std_indices:
                    src_state = copy.deepcopy(self.experts[std_indices[0]].state_dict())
                    for idx in self._harmful_indices:
                        self.experts[idx].load_state_dict(src_state)

        def forward(self, x: torch.Tensor) -> torch.Tensor:
            B, T, C = x.size()
            x_flat = x.view(-1, C)
            router = self.gate(x_flat)
            probs, indices = torch.topk(router, self.config.n_expert_per_token)
            probs = probs.softmax(dim=1, dtype=torch.float).to(dtype=x_flat.dtype)
            masks = indices.unsqueeze(-1) == torch.arange(self.config.n_expert, device=x_flat.device)
            masks = masks.permute(2, 0, 1)
            y = torch.zeros_like(x_flat)
            for expert_idx, (mask, expert) in enumerate(zip(masks, self.experts)):
                if self._activation_masking_enabled and expert_idx in self._harmful_indices:
                    continue  # skip harmful expert — zero contribution
                token_idx, sel_expert_idx = torch.where(mask)
                if token_idx.numel() == 0:
                    continue
                y[token_idx] += probs[token_idx, sel_expert_idx, None] * expert(x_flat[token_idx])
            return y.view(B, T, C)
    ```
